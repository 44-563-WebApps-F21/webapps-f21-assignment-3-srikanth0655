    double area(const vector& fig) {
  double res = 0;
  for (unsigned i = 0; i < fig.size(); i++) {
      point p = i ? fig[i - 1] : fig.back();
      point q = fig[i];
      res += (p.x - q.x) * (p.y + q.y);
  }
  return fabs(res) / 2;
}
